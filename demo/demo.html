      $(function(){
        $(document).observe_items({
          singleItems:["multiply", "divisor"],
          multipleItems:["add","subtract"], 
          callback : calculate_total
        });
      });


      function calculate_total(params){
        var total = 0;
        var items = params["add"];
        for ( var i in items){
          total += items[i];
        }
        items = params["subtract"];
        for (var i in items){
          total -= items[i];
        }
        total *= params["multiply"];
        total /= params["divisor"];
        $("#total").html(total);
      }
